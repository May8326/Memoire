\newglossaryentry{thesaurus}{
	name=thésaurus,
	sort=thesaurus,
	description={Outil de représentation des connaissances sous forme de vocabulaire contrôlé, structuré selon des relations sémantiques et hiérarchiques entre les concepts. Un thésaurus permet d’organiser les termes utilisés pour indexer, rechercher et retrouver l’information dans des systèmes documentaires. Il repose sur des relations telles que l’équivalence (synonymes), la hiérarchie (termes génériques/spécifiques) et l’association (liens contextuels), facilitant ainsi la normalisation du langage documentaire et l’interopérabilité entre bases de données. Contrairement à un simple lexique, le thésaurus vise à refléter une vision cohérente d’un domaine de connaissance. Pour plus de précision, voir la section \refinterne{III-A-2.1}}
}

\newglossaryentry{skos}{
	name=\ac{skos},
	sort=skos,
	description={Modèle de représentation standardisé développé par le \ac{w3c} pour structurer et publier des vocabulaires contrôlés tels que des thésaurus, des taxonomies, des classifications ou des listes d’autorité dans le cadre du Web sémantique. SKOS repose sur les principes de \ac{rdf} et permet d’exprimer les relations conceptuelles (hiérarchiques, associatives, équivalences) entre les termes, facilitant ainsi l’interopérabilité, la réutilisation et la découverte des ressources numériques dans des environnements hétérogènes.}
}

\newglossaryentry{rdf}{
	name=\ac{rdf},
	sort=rdf,
	description={Cadre de modélisation des données développé par le \ac{w3c} pour représenter l'information sur le Web de manière structurée et interopérable. RDF repose sur une logique de triplets : sujet – prédicat – objet, permettant de décrire les relations entre les ressources. Chaque élément est identifié par une URI (\textit{Uniform Resource Identifier} -- littéralement identifiant uniforme de ressource), ce qui facilite le partage, la réutilisation et l’intégration des données à grande échelle. RDF est la base du Web sémantique, et sert de fondation à des langages comme OWL ou SKOS.}
}

\newglossaryentry{theorie3ages}{
	name=théorie des trois âges,
	sort=théorie des trois âges,
	description={Notion fondamentale sur laquelle repose l'archivistique
		contemporaine, et qui fait passer tout document par trois périodes,
		courante, intermédiaire et définitive, caractérisées par la fréquence
		et le type d'utilisation qui en est faite. Voir \gls{archivecourante}, \gls{archiveintermediaire}, \gls{archivedefinitive}\footcite{directiondesarchivesdefranceDictionnaireTerminologieArchivistique2002}}
}

\newglossaryentry{archivecourante}{
	name=archives courantes,
	description={Documents qui sont d'utilisation habituelle et fréquente pour l'activité des services, établissements et organismes qui les ont produits et reçus, et qui sont conservés pour le traitement des affaires\footcite{directiondesarchivesdefranceDictionnaireTerminologieArchivistique2002}}
}

\newglossaryentry{archiveintermediaire}{
	name=archives intermédiaires,
	description={Documents qui, n'étant plus
		d'usage courant, doivent être conservés temporairement, pour des
		besoins administratifs ou juridiques (y compris les documents qui
		après tri seront conservés comme des archives définitives)\footcite{directiondesarchivesdefranceDictionnaireTerminologieArchivistique2002}}
}

\newglossaryentry{archivedefinitive}{
	name=archives définitives,
	description={Documents qui, ayant subi des
		tris, ne sont plus susceptibles d'élimination, par opposition aux
		archives courantes ou intermédiaires, et qui sont conservés pour les
		besoins de la gestion et de la justification des droits des personnes
		physiques ou morales, publiques ou privées, et pour la
		documentation historique de la recherche\footcite{directiondesarchivesdefranceDictionnaireTerminologieArchivistique2002}}
}
